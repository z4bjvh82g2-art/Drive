import pygame import math import random pygame.init() WIDTH, HEIGHT = 1000, 700 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Racing Game Ultimate") clock = pygame.time.Clock() font = pygame.font.SysFont("Arial", 20) # Strecke waypoints = [ (200,300),(400,200),(700,250),(800,400), (700,550),(400,600),(200,500),(150,400) ] brake_zones = [1, 3, 5] # ========================= # 🚗 Car Klasse # ========================= class Car: def __init__(self, x, y, color, is_ai=False): self.x = x self.y = y self.angle = 0 # Physik self.speed = 0 self.velocity_x = 0 self.velocity_y = 0 self.acc = 0.3 self.max_speed = 7 self.grip = 0.9 # Motor self.rpm = 1000 self.gear = 1 # Schaden self.health = 100 # KI self.target_wp = 0 self.is_ai = is_ai self.color = color def update_engine(self): self.rpm = 1000 + abs(self.speed) * 900 if self.rpm > 7000 and self.gear < 5: self.gear += 1 self.rpm = 4000 elif self.rpm < 2000 and self.gear > 1: self.gear -= 1 self.rpm = 3000 def apply_physics(self): # Richtungsvektor forward_x = math.cos(math.radians(self.angle)) forward_y = -math.sin(math.radians(self.angle)) # Drift (seitliche Bewegung reduzieren) self.velocity_x = self.velocity_x * self.grip + forward_x * self.speed * (1 - self.grip) self.velocity_y = self.velocity_y * self.grip + forward_y * self.speed * (1 - self.grip) self.x += self.velocity_x self.y += self.velocity_y # Reibung self.speed *= 0.98 def take_damage(self, amount): self.health -= amount self.health = max(0, self.health) # Einfluss auf Fahrverhalten self.max_speed = 7 * (self.health / 100) self.acc = 0.3 * (self.health / 100) def draw(self): surf = pygame.Surface((40,20)) surf.fill(self.color) rotated = pygame.transform.rotate(surf, self.angle) rect = rotated.get_rect(center=(self.x, self.y)) screen.blit(rotated, rect.topleft) # ========================= # 🤖 KI # ========================= def ai_drive(car, player): target = waypoints[car.target_wp] dx = target[0] - car.x dy = target[1] - car.y target_angle = math.degrees(math.atan2(-dy, dx)) diff = (target_angle - car.angle + 180) % 360 - 180 car.angle += diff * 0.07 # Bremszone if car.target_wp in brake_zones: car.speed *= 0.94 else: car.speed += car.acc # Überholen dist = math.hypot(car.x - player.x, car.y - player.y) if dist < 70: car.angle += random.choice([-25, 25]) car.speed += 0.4 car.speed = min(car.speed, car.max_speed) # Bewegung car.apply_physics() if math.hypot(dx, dy) < 40: car.target_wp = (car.target_wp + 1) % len(waypoints) # ========================= # Spieler & Gegner # ========================= player = Car(300,400,(255,0,0)) ai_cars = [ Car(320,420,(0,0,255),True), Car(340,440,(0,255,0),True) ] # ========================= # Kollisionen # ========================= def check_collisions(player, ai_cars): for ai in ai_cars: dist = math.hypot(player.x - ai.x, player.y - ai.y) if dist < 40: player.take_damage(5) ai.take_damage(5) # Rückstoß player.speed *= -0.5 ai.speed *= -0.5 # ========================= # UI # ========================= def draw_ui(): screen.blit(font.render(f"RPM: {int(player.rpm)}", True, (255,255,255)), (10,10)) screen.blit(font.render(f"Gear: {player.gear}", True, (255,255,255)), (10,30)) screen.blit(font.render(f"Health: {int(player.health)}%", True, (255,255,255)), (10,50)) # ========================= # Game Loop # ========================= running = True while running: screen.fill((30,150,30)) pygame.draw.lines(screen,(200,200,200),True,waypoints,50) for event in pygame.event.get(): if event.type == pygame.QUIT: running = False keys = pygame.key.get_pressed() # Steuerung if keys[pygame.K_UP]: player.speed += player.acc if keys[pygame.K_DOWN]: player.speed -= player.acc if keys[pygame.K_LEFT]: player.angle += 4 if keys[pygame.K_RIGHT]: player.angle -= 4 player.speed = max(-2, min(player.speed, player.max_speed)) player.update_engine() player.apply_physics() # KI for ai in ai_cars: ai_drive(ai, player) ai.update_engine() ai.draw() # Kollisionen check_collisions(player, ai_cars) # Zeichnen player.draw() draw_ui() pygame.display.flip() clock.tick(60) pygame.quit()
